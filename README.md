    # An integer sequence
    vector_var <- 2:6 # 2 3 4 5 6
    
    # A complex sequence
    vector_var <- seq(2, 3, by=0.5) # 2.0 2.5 3.0 

    # Repeat a vector
    vector_var <- rep(1:2, times=3) # 1 2 1 2 1 2

    # Repeat elements
    vector_var <- rep(1:2, each=3) # 1 1 1 2 2 2
